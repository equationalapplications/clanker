# LLM Wiki Phase 1 — Package Upgrade Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Upgrade `@equationalapplications/expo-llm-wiki` from `^3.0.0` to `4.1.0`, audit the v3→v4 migration on a populated DB, and update `wikiService` config to mobile-first defaults — without touching call sites or behavior beyond the new defaults.

**Architecture:** Single PR. Bump dep, run `wiki.setup()` against a migration fixture to confirm no data loss, then update the `setupWiki` config object. All existing call sites (`useMemoryRead`, `useWikiWrite`, `useWikiIngest`, `useWikiForget`, `useWikiExport`, `useWikiMaintenance`, `useWikiHasChanged`, `useWiki`, `WikiProvider`, `WikiBusyError`, `formatContext`, `MemoryDump`, `createWiki`) remain on the package's public API; only re-verify they still type-check post-bump. State machine, orchestrator, and gating changes land in later phases.

**Tech Stack:** React Native (Expo), TypeScript, `expo-sqlite`, Jest, `@equationalapplications/expo-llm-wiki@4.1.0`.

**Spec:** `docs/superpowers/specs/2026-05-09-llm-wiki-state-machine-design.md` § Phase 1.

/**
 * Migration audit: v3 schema seeded into SQLite, then wiki@4.1.0 setup() runs.
 * Asserts setup() does not throw and pre-existing rows are preserved.
 *
 * This test guards the v3 → v4 upgrade per spec § Risks.
 */
import { createWiki } from '@equationalapplications/expo-llm-wiki'
import { openDatabaseSync, type SQLiteDatabase } from 'expo-sqlite'

// Minimal v3 schema fragment — table prefix matches wikiService.
const V3_TABLES_SQL = `
  CREATE TABLE IF NOT EXISTS llm_wiki_entities (
    id TEXT PRIMARY KEY,
    name TEXT,
    created_at INTEGER NOT NULL
  );
  CREATE TABLE IF NOT EXISTS llm_wiki_facts (
    id TEXT PRIMARY KEY,
    entity_id TEXT NOT NULL,
    text TEXT NOT NULL,
    created_at INTEGER NOT NULL
  );
`

function seedV3(db: SQLiteDatabase) {
  db.execSync(V3_TABLES_SQL)
  db.runSync(
    `INSERT INTO llm_wiki_entities (id, name, created_at) VALUES (?, ?, ?)`,
    ['ent-1', 'Test Char', 1_700_000_000_000],
  )
  db.runSync(
    `INSERT INTO llm_wiki_facts (id, entity_id, text, created_at) VALUES (?, ?, ?, ?)`,
    ['fact-1', 'ent-1', 'likes coffee', 1_700_000_001_000],
  )
}

jest.mock('~/services/wikiLlmProvider', () => ({
  createWikiLlmProvider: () => ({ generateText: jest.fn() }),
}))

describe('wiki v3 → v4 migration audit', () => {
  let db: SQLiteDatabase

  beforeEach(() => {
    db = openDatabaseSync(':memory:')
    seedV3(db)
  })

  afterEach(() => {
    db.closeSync()
  })

  it('setup() completes without error against v3-seeded DB', async () => {
    const wiki = createWiki(db, {
      llmProvider: { generateText: jest.fn() } as any,
      config: { tablePrefix: 'llm_wiki_' },
    })
    await expect(wiki.setup()).resolves.not.toThrow()
  })

  it('preserves pre-existing entity row after migration', async () => {
    const wiki = createWiki(db, {
      llmProvider: { generateText: jest.fn() } as any,
      config: { tablePrefix: 'llm_wiki_' },
    })
    await wiki.setup()
    const row = db.getFirstSync<{ id: string; name: string }>(
      `SELECT id, name FROM llm_wiki_entities WHERE id = ?`,
      ['ent-1'],
    )
    expect(row).toEqual({ id: 'ent-1', name: 'Test Char' })
  })

  it('preserves pre-existing fact row after migration', async () => {
    const wiki = createWiki(db, {
      llmProvider: { generateText: jest.fn() } as any,
      config: { tablePrefix: 'llm_wiki_' },
    })
    await wiki.setup()
    const row = db.getFirstSync<{ id: string; text: string }>(
      `SELECT id, text FROM llm_wiki_facts WHERE id = ?`,
      ['fact-1'],
    )
    expect(row).toEqual({ id: 'fact-1', text: 'likes coffee' })
  })
})
```

// src/services/wikiService.ts
import { createWiki } from '@equationalapplications/expo-llm-wiki'
import type { SQLiteDatabase } from 'expo-sqlite'
import { createWikiLlmProvider } from './wikiLlmProvider'

type Wiki = ReturnType<typeof createWiki>

let _wiki: Wiki | null = null

export function setupWiki(db: SQLiteDatabase): Wiki {
  if (_wiki) return _wiki
  _wiki = createWiki(db, {
    llmProvider: createWikiLlmProvider(),
    config: {
      tablePrefix: 'llm_wiki_',
      autoLibrarianThreshold: 5,
      autoHealThreshold: 100,
      pruneRetainSoftDeletedFor: 3,
      pruneEventsAfter: 14,
      orphanAfterDays: 14,
      staleInferredAfterDays: 30,
      preFilterLimit: 300,
      hybridWeight: 0.7,
    },
  })
  return _wiki
}

export function getWiki(): Wiki | null {
  return _wiki
}

export async function initWiki(db: SQLiteDatabase): Promise<void> {
  const wiki = setupWiki(db)
  await wiki.setup()
}

/** For tests only — reset the singleton between test runs. */
export function _resetWikiForTests(): void {
  _wiki = null
}
```
